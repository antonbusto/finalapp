import React from 'react';
import { IonPage, IonHeader, IonToolbar, IonTitle, IonContent, IonImg, IonText } from "@ionic/react";
import { useParams } from "react-router-dom";
import moviesData from "../../data/movies.json";
import MenuPrincipal from '../../components/MenuPrincipal/MenuPrincipal';

interface Pelicula {
  id: number;
  title: string;
  poster_path: string;
  overview: string;
  release_date: string;
  vote_average: number;
}

interface RouteParams {
  id: string;
}

const Pelicula: React.FC = () => {
  const { id } = useParams<RouteParams>();
  const pelicula = moviesData.find((movie: Pelicula) => movie.id.toString() === id);

  if (!pelicula) {
    return (
      <IonPage>
        <IonHeader>
          <IonToolbar color="primary">
            <IonTitle>Película no encontrada</IonTitle>
          </IonToolbar>
        </IonHeader>
        <IonContent>
          <IonText>Lo sentimos, no pudimos encontrar esta película.</IonText>
        </IonContent>
      </IonPage>
    );
  }

  const { title, poster_path, overview, release_date, vote_average } = pelicula;
  const imageUrl = `https://image.tmdb.org/t/p/w500${poster_path}`;

  return (
    <IonPage>
      <IonHeader>
        <IonToolbar color="primary">
          <IonTitle>{title}</IonTitle>
        </IonToolbar>
      </IonHeader>
      <IonContent className="pelicula-content">
        <IonImg src={imageUrl} alt={title} />
        <IonText>
          <h2>{title}</h2>
        </IonText>
        <IonText>
          <p><strong>Estreno:</strong> {release_date}</p>
        </IonText>
        <IonText>
          <p><strong>Calificación:</strong> {vote_average}</p>
        </IonText>
        <IonText>
          <p>{overview}</p>
        </IonText>
      </IonContent>
    </IonPage>
  );
};

export default Pelicula;
